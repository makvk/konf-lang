let default_port = 8080
let ssl_port = 8443
let hosts = list("example.com", "api.example.com", "static.example.com")

dict(
    server = dict(
        port = !{default_port},
        ssl_port = !{ssl_port},
        timeout = 30
    ),
    hosts = !{sort(hosts)},
    logging = dict(
        level = "info",
        format = "json"
    )
)
